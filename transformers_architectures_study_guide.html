1. Baseline: original Transformer (Vaswani et al.)

The paper defines an encoder stack and a decoder stack. Each encoder layer has multi-head self-attention (bidirectional over the source), then a position-wise FFN, with residuals and layer normalization (exact norm placement varies by implementation; the paper uses post-LN). Each decoder layer adds masked self-attention (causal) and cross-attention to encoder outputs. Positions use sinusoidal encodings added to embeddings.

Encoder layer (calls §1.1; post-norm like original paper)

def encoder_layer(x, pad_mask, n_heads, ln1_g, ln1_b, ln2_g, ln2_b, attn, ff):
    # x: (batch, seq, dim); pad_mask: optional additive mask for padding (see §1.1)
    h = multi_head_self_attention(x, attn, n_heads, attn_mask=pad_mask)
    x = layer_norm(x + h, ln1_g, ln1_b)
    h = feed_forward(x, ff)
    return layer_norm(x + h, ln2_g, ln2_b)

Decoder layer (same primitives + causal self-attn + cross-attn)

def decoder_layer(x, encoder_out, causal, enc_key_pad_mask, n_heads,
                  ln1_g, ln1_b, ln2_g, ln2_b, ln3_g, ln3_b, self_attn, cross_attn, ff):
    # causal: (1,1,T_dec,T_dec) from causal_mask(); enc_key_pad_mask: mask encoder keys (pads)
    h = masked_multi_head_self_attention(x, self_attn, n_heads, causal, extra_mask=None)
    x = layer_norm(x + h, ln1_g, ln1_b)
    h = multi_head_cross_attention(x, encoder_out, cross_attn, n_heads, attn_mask=enc_key_pad_mask)
    x = layer_norm(x + h, ln2_g, ln2_b)
    h = feed_forward(x, ff)
    return layer_norm(x + h, ln3_g, ln3_b)

1.1 Building-block implementations (stacked references)

These snippets are minimal PyTorch-style sketches. Order matches dependency: layer_norm and feed_forward are independent; attention builds on a shared _scaled_dot_product_attention; multi_head_self_attention is used by the encoder; masked_multi_head_self_attention wraps it with a causal mask; multi_head_cross_attention reuses the same MHA core with Q from the decoder and K,V from encoder memory.

(1) layer_norm — normalize over hidden dim (post-norm residual path)

import math
import torch
import torch.nn.functional as F

def layer_norm(x, gamma, beta, eps=1e-5):
    """x: (batch, seq, dim). gamma, beta: (dim,) — learnable scale & shift."""
    mean = x.mean(dim=-1, keepdim=True)
    var = x.var(dim=-1, unbiased=False, keepdim=True)
    x_hat = (x - mean) / torch.sqrt(var + eps)
    return gamma * x_hat + beta

(2) feed_forward — two linear layers with nonlinearity (position-wise)

def feed_forward(x, ff):
    """ff holds weights/bias for W1: dim→ff_dim and W2: ff_dim→dim."""
    h = F.linear(x, ff.w1, ff.b1)
    h = F.gelu(h)  # paper used ReLU; BERT/GPT often GELU
    return F.linear(h, ff.w2, ff.b2)

(3) Shared core — split heads, merge heads, scaled dot-product

def _split_heads(t, n_heads):
    """(B, T, D) → (B, n_heads, T, d_head) where D = n_heads * d_head"""
    B, T, D = t.shape
    d_head = D // n_heads
    t = t.view(B, T, n_heads, d_head).transpose(1, 2)
    return t

def _merge_heads(t):
    """(B, n_heads, T, d_head) → (B, T, D)"""
    B, H, T, dh = t.shape
    return t.transpose(1, 2).contiguous().view(B, T, H * dh)

def _scaled_dot_product_attention(q, k, v, attn_mask=None):
    """q,k,v: (B, n_heads, T_q or T_k, d_head). attn_mask: broadcastable, additive (0 or -inf)."""
    d_head = q.size(-1)
    scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(d_head)
    if attn_mask is not None:
        scores = scores + attn_mask
    weights = F.softmax(scores, dim=-1)
    return torch.matmul(weights, v)

(4) multi_head_self_attention — Q, K, V all from x (encoder / bidirectional)

def multi_head_self_attention(x, attn, n_heads, attn_mask=None):
    """attn: packed Q/K/V output projections + output projection W_o."""
    B, T, D = x.shape
    q = F.linear(x, attn.w_q, attn.b_q)
    k = F.linear(x, attn.w_k, attn.b_k)
    v = F.linear(x, attn.w_v, attn.b_v)
    q, k, v = _split_heads(q, n_heads), _split_heads(k, n_heads), _split_heads(v, n_heads)
    # Optional padding mask as (B,1,1,T) large negative at pad positions
    ctx = _scaled_dot_product_attention(q, k, v, attn_mask=attn_mask)
    ctx = _merge_heads(ctx)
    return F.linear(ctx, attn.w_o, attn.b_o)

(5) masked_multi_head_self_attention — same as (4) + causal mask

def causal_mask(T, device, dtype):
    """Additive mask: allowed=0, disallowed=-inf (upper triangle). Shape (1,1,T,T)."""
    m = torch.triu(torch.full((T, T), float("-inf"), device=device, dtype=dtype), diagonal=1)
    return m.view(1, 1, T, T)

def masked_multi_head_self_attention(x, attn, n_heads, causal_only_mask, extra_mask=None):
    """Decoder self-attn: causal_only_mask from causal_mask(T,...); add padding via extra_mask."""
    mask = causal_only_mask
    if extra_mask is not None:
        mask = mask + extra_mask  # combine padding + causal
    return multi_head_self_attention(x, attn, n_heads, attn_mask=mask)

(6) multi_head_cross_attention — Q from decoder state, K,V from encoder memory

def multi_head_cross_attention(x, memory, attn, n_heads, attn_mask=None):
    """x: (B, T_dec, D); memory: (B, T_enc, D). attn_mask often masks pad keys (enc padding)."""
    q = F.linear(x, attn.w_q, attn.b_q)
    k = F.linear(memory, attn.w_k, attn.b_k)
    v = F.linear(memory, attn.w_v, attn.b_v)
    q, k, v = _split_heads(q, n_heads), _split_heads(k, n_heads), _split_heads(v, n_heads)
    ctx = _scaled_dot_product_attention(q, k, v, attn_mask=attn_mask)
    ctx = _merge_heads(ctx)
    return F.linear(ctx, attn.w_o, attn.b_o)

1.2 Call graph (how the pieces stack)

Encoder layer (top of §1) = multi_head_self_attention → residual → layer_norm → feed_forward → residual → layer_norm. Decoder layer = masked_multi_head_self_attention (wraps multi_head_self_attention + causal mask) → layer_norm → multi_head_cross_attention (same _scaled_dot_product_attention core, different Q/K/V sources) → layer_norm → feed_forward → layer_norm.

# Dependency summary (all defined in §1.1):
# layer_norm ─────────────────────────────┐
# feed_forward ──────────────────────────┤ used by encoder & decoder layers
# multi_head_self_attention ──────────────┤
#    ↑ called by masked_multi_head_self_attention
# multi_head_cross_attention ─────────────┘ (shares _scaled_dot_product_attention with self-attn)

Hugging Face does not expose one single class as “the baseline” for all models. Instead, each folder under models/ implements a paper’s variant. Use §1.1–1.2 as the mental diff target when reading HF models/.

RoPE idea (applied to Q and K after projection)

# Simplified: rotate pairs of dimensions of q, k by position-dependent angles
def apply_rotary_pos_emb(q, k, cos, sin):
    # q, k: (batch, heads, seq, head_dim)
    # cos, sin: broadcastable to q, k
    q_embed = (q * cos) + (rotate_half(q) * sin)
    k_embed = (k * cos) + (rotate_half(k) * sin)
    return q_embed, k_embed

SwiGLU MLP (compare to single-gate FFN)

class SwiGLU(nn.Module):
    def __init__(self, dim, hidden_dim):
        super().__init__()
        self.gate_proj = nn.Linear(dim, hidden_dim, bias=False)
        self.up_proj = nn.Linear(dim, hidden_dim, bias=False)
        self.down_proj = nn.Linear(hidden_dim, dim, bias=False)

    def forward(self, x):
        return self.down_proj(F.silu(self.gate_proj(x)) * self.up_proj(x))

# Conceptual: num_heads=32, num_key_value_heads=8
# After projecting K,V with 8 heads, expand/repeat to 32 for attention broadcast
def repeat_kv(hidden_states, n_rep):
    if n_rep == 1:
        return hidden_states
    batch, num_kv_heads, slen, head_dim = hidden_states.shape
    hidden_states = hidden_states[:, :, None, :, :].expand(batch, num_kv_heads, n_rep, slen, head_dim)
    return hidden_states.reshape(batch, num_kv_heads * n_rep, slen, head_dim)

4. Curated architecture comparisons

Each block: HF module path, vs parent, removed / changed / added, purpose, plus inline code.

BERT (models/bert/)

Vs full Transformer: encoder-only.

  • Removed: entire decoder; cross-attention; autoregressive head (replaced by MLM/NSP training heads in original BERT).
  • Changed: self-attention is bidirectional (no causal mask on encoder).
  • Added: [CLS], [SEP]; token type / segment embeddings; learned position embeddings.
  • Purpose: Deep contextual representations for NLU; sentence-pair tasks via types.
# Embedding sum typical of BERT-style models
def embed_bert(input_ids, token_type_ids):
    tok = word_embeddings(input_ids)
    pos = position_embeddings(position_ids)  # learned table
    seg = token_type_embeddings(token_type_ids)
    return layer_norm(tok + pos + seg)

# No causal mask in self-attention for standard BERT encoder
# attn_scores visible to full sequence (plus padding mask only)

RoBERTa (models/roberta/)

Vs BERT: same encoder architecture family; training recipe differs.

  • Removed: NSP objective (next sentence prediction) in the RoBERTa recipe.
  • Changed: dynamic masking, larger batches, byte-level BPE—mostly training, not a new layer type.
  • Purpose: Better NLU with simpler objectives and more data/compute.
# Modeling stack mirrors BERT-like encoder; difference is not a new op:
# RoBERTaForMaskedLM(...) still uses RobertaEmbeddings + RobertaLayer x L
# Key idea: only MLM (and sometimes sequence-level tasks), no NSP loss in RoBERTa pretraining

GPT-2 (models/gpt2/)

Vs full Transformer: decoder-only.

  • Removed: encoder; cross-attention.
  • Changed: causal self-attention; pre-LayerNorm stack in GPT-2.
  • Added: learned position embeddings; single LM head tied or untied to embeddings.
  • Purpose: Unconditional and conditional text generation via left-to-right modeling.
def causal_attention_mask(seq_len, device):
    # prevent attending to future positions
    mask = torch.full((seq_len, seq_len), float("-inf"), device=device)
    mask = torch.triu(mask, diagonal=1)
    return mask  # added to logits before softmax

def gpt2_block(x, attn, mlp, ln_1, ln_2):
    x = x + attn(ln_1(x))   # pre-norm
    x = x + mlp(ln_2(x))
    return x

T5 (models/t5/)

Vs vanilla Transformer: still encoder–decoder, but relative position biases inside attention and text-to-text framing.

  • Changed: position info often via learned bias per head and relative bucket (not additive sinusoidal on embeddings only).
  • Added: task prefixes as text (“translate: …”)—data/API convention; architecture is seq2seq.
  • Purpose: Unified supervised objective across many NLP tasks.
# Conceptual: attention logits get a bias from relative position bucket
scores = (q @ k.transpose(-1, -2)) / math.sqrt(d)
scores = scores + compute_bias(relative_position_bucket)  # shape broadcast to scores
probs = softmax(scores, dim=-1)

BART (models/bart/)

Vs T5: also encoder–decoder; BART uses Transformer-style absolute sinusoidal or learned configs depending on variant; denoising pretraining (corrupt document, reconstruct).

  • Changed: Encoder is often bidirectional; decoder is autoregressive with cross-attn—classic seq2seq Transformer pattern.
  • Purpose: Strong seq2seq for summarization, dialogue, etc., with denoising objective.
def bart_forward(encoder_ids, decoder_ids, encoder_padding_mask):
    enc = encoder(input_ids=encoder_ids, attention_mask=encoder_padding_mask)
    dec = decoder(
        input_ids=decoder_ids,
        encoder_hidden_states=enc.last_hidden_state,
        encoder_attention_mask=encoder_padding_mask,
    )
    return dec.last_hidden_state

DistilBERT (models/distilbert/)

Vs BERT: shallower (fewer layers), knowledge distillation from a teacher BERT.

  • Removed: token type embeddings in DistilBERT (single sentence); pooler for NSP not used same way.
  • Changed: ~half the layers; smaller/faster inference.
  • Purpose: Speed and size with minimal accuracy drop.
# Fewer Transformer layers; same per-layer pattern as BERT-like encoder
# Student forward matches teacher logits/hidden states during distillation (training)
for i in range(config.num_hidden_layers):  # e.g. 6 vs 12
    hidden_states = transformer_layer(hidden_states, attn_mask)

OPT (models/opt/)

Vs GPT-2: decoder-only LM; similar pre-norm and learned positions; scales to larger models.

  • Changed: architecture details (layer norm epsilon, init, bias flags) per OPT config.
  • Purpose: Open reproduction of large decoder LMs.
# OPTDecoderLayer: self-attn + FFN, residual, LayerNorm before each sublayer
def opt_decoder_layer(hidden_states, attention_mask, layer):
    residual = hidden_states
    hidden_states = layer.self_attn_layer_norm(hidden_states)
    hidden_states, _ = layer.self_attn(hidden_states, attention_mask=attention_mask)
    hidden_states = residual + hidden_states
    residual = hidden_states
    hidden_states = layer.final_layer_norm(hidden_states)
    hidden_states = layer.fc1(hidden_states)
    hidden_states = activation_fn(hidden_states)
    hidden_states = layer.fc2(hidden_states)
    return residual + hidden_states

DeBERTa / DeBERTa-v2 (models/deberta, deberta_v2)

Vs BERT: encoder-only but disentangled attention (separate content vs relative position) and enhanced mask decoder in v2.

  • Added: relative position as separate keys; content-to-relative interaction.
  • Changed: v2 uses convolution or bucketed structures in some components (see config); stronger on long contexts and GLUE-style tasks.
  • Purpose: Richer pairwise modeling than standard dot-product + absolute positions only.
# Conceptual disentangled idea: two streams for c (content) and r (relative)
# Attention scores combine interactions between content and relative position embeddings
scores = (
    (q_c @ k_c.transpose(-1, -2))
    + (q_c @ k_r.transpose(-1, -2))
    + (q_r @ k_c.transpose(-1, -2))
    + (q_r @ k_r.transpose(-1, -2))
) / math.sqrt(d)

ViT (models/vit/)

Vs NLP Transformer: images → patch tokens; encoder stack; CLS or pooling for classification.

  • Removed: decoder (for classification ViT); no text tokenizer.
  • Added: Patch embedding (Conv2d with stride = patch size); optional [CLS] token.
  • Purpose: Apply Transformer to vision without hand-crafted CNN stem (still patch projection).
def patchify(images, patch_size, proj):  # images: (B, 3, H, W)
    # Conv2d(in_ch, dim, kernel=patch_size, stride=patch_size) -> (B, dim, H', W')
    x = proj(images)
    x = x.flatten(2).transpose(1, 2)  # (B, num_patches, dim)
    return x

def vit_encoder(x, cls_token, pos_embed, layers):
    x = torch.cat([cls_token.expand(x.size(0), -1, -1), x], dim=1)
    x = x + pos_embed
    for layer in layers:
        x = layer(x)
    return x

Swin (models/swin/)

Vs ViT: hierarchical feature maps + shifted window self-attention.

  • Added: attention inside local windows; cyclic shift across blocks for cross-window connections.
  • Changed: patch merging stages reduce resolution → CNN-like pyramid.
  • Purpose: Linear-ish complexity in image size vs global ViT attention cost.
def window_partition(x, window_size):
    # x: (B, H, W, C) -> windows
    B, H, W, C = x.shape
    x = x.view(B, H // window_size, window_size, W // window_size, window_size, C)
    windows = x.permute(0, 1, 3, 2, 4, 5).contiguous().view(-1, window_size, window_size, C)
    return windows

# Alternate blocks shift windows to allow information flow between windows

CLIP (models/clip/)

Vs single-stack LM: two encoders (image + text) and contrastive alignment in embedding space.

  • Removed: autoregressive decoder for the core CLIP model (generation is not the pretrained objective).
  • Added: dual towers; logit scale temperature; cosine similarity / contrastive loss training.
  • Purpose: Zero-shot classification and multimodal retrieval via shared embedding geometry.
def clip_logits(image_embeds, text_embeds, logit_scale):
    # L2-normalize
    image_embeds = image_embeds / image_embeds.norm(dim=-1, keepdim=True)
    text_embeds = text_embeds / text_embeds.norm(dim=-1, keepdim=True)
    logits = (image_embeds @ text_embeds.T) * logit_scale.exp()
    return logits  # (batch_img, batch_txt) similarity matrix

BLIP-2 (models/blip_2/)

Vs CLIP: adds a Q-Former (lightweight transformer) between frozen image encoder and frozen LLM.

  • Added: learnable query tokens attending to image features; projection into LLM hidden size.
  • Purpose: Cheap adaptation of powerful frozen models for VQA / captioning without full finetune of huge backbones.
def qformer_forward(query_tokens, encoder_hidden_states, cross_attn_layers):
    # query_tokens: learned embeddings (B, num_queries, D)
    x = query_tokens
    for layer in cross_attn_layers:
        x = layer.self_attn(x)
        x = layer.cross_attn(x, context=encoder_hidden_states)  # attend to image tokens
        x = layer.mlp(x)
    return x  # compressed visual tokens -> feed LLM as soft prompts

LLaMA (models/llama/)

Vs GPT-2: decoder-only but RMSNorm, RoPE, SwiGLU, typically no bias in linear layers; often GQA in larger variants.

  • Changed: norm type; position handling; MLP shape.
  • Purpose: Efficient, high-quality open-weights LLM inference and finetuning ecosystem.
class LlamaDecoderLayer(nn.Module):
    def forward(self, hidden_states, attention_mask, position_ids, past_key_value):
        residual = hidden_states
        hidden_states = self.input_layernorm(hidden_states)
        hidden_states, present_kv = self.self_attn(
            hidden_states=hidden_states,
            attention_mask=attention_mask,
            position_ids=position_ids,
            past_key_value=past_key_value,
        )
        hidden_states = residual + hidden_states
        residual = hidden_states
        hidden_states = self.post_attention_layernorm(hidden_states)
        hidden_states = self.mlp(hidden_states)
        return residual + hidden_states, present_kv

Mistral (models/mistral/)

Vs LLaMA: same broad decoder + RoPE + RMSNorm + SwiGLU family; adds sliding-window attention on many layers.

  • Added / changed: local attention window (global layers optional in some configs).
  • Purpose: Long context efficiency with strong quality per parameter.
def sliding_window_mask(seq_len, window, device):
    # True = allow attend; causal + only last `window` positions
    mask = torch.ones(seq_len, seq_len, device=device, dtype=torch.bool).tril()
    for i in range(seq_len):
        mask[i, : max(0, i - window + 1)] = False
    return mask

Mixtral (MoE) (models/mixtral/)

Vs dense LLaMA-like decoder: FFN replaced by sparse mixture-of-experts.

  • Added: router logits per token; top-k expert selection; per-expert feed-forward.
  • Purpose: Increase parameter count (capacity) while activating only a subset per token (compute).
def moe_ffn(x, gate, experts, top_k=2):
    # x: (batch*seq, dim)
    router_logits = gate(x)  # (batch*seq, num_experts)
    weights, selected = torch.topk(router_logits, k=top_k, dim=-1)
    weights = torch.softmax(weights, dim=-1)
    out = torch.zeros_like(x)
    for i, expert in enumerate(experts):
        mask = (selected == i).any(dim=-1)
        if mask.any():
            out[mask] += expert(x[mask])  # simplified; real code uses batched routing
    return out

Longformer (models/longformer/)

Vs BERT: encoder with local window + global tokens (e.g. [CLS]) attending everywhere.

  • Changed: attention pattern is sparse (not full n×n for all tokens).
  • Purpose: Long documents with manageable memory.
# Pattern: diagonal band (local) + special global indices attend to all / attended by all
# Implementation uses custom CUDA/CPU kernels in HF for efficiency
def longformer_pattern(seq_len, window, global_indices):
    allowed = torch.zeros(seq_len, seq_len, dtype=torch.bool)
    for i in range(seq_len):
        lo, hi = max(0, i - window // 2), min(seq_len, i + window // 2 + 1)
        allowed[i, lo:hi] = True
    for g in global_indices:
        allowed[g, :] = True
        allowed[:, g] = True
    return allowed

BigBird (models/big_bird/)

Vs BERT: random + window + global sparse attention pattern (theoretical approximator of full attention).

  • Added: random attention edges; blockwise structure.
  • Purpose: Long sequence encoder with scalability.
# Three-part pattern per query block (conceptual):
# - attend to nearby tokens (window)
# - attend to a few random keys (approximation)
# - global tokens (e.g. first few) attend bidirectionally to all
# HF implements this with block-sparse attention kernels, not a dense n x n matrix.
def bigbird_attention_allowed(i, j, window, random_partners, global_idx):
    if j in global_idx or i in global_idx:
        return True
    if abs(i - j) <= window:
        return True
    return j in random_partners[i]

Whisper (models/whisper/)

Vs text-only: audio encoder (Conv front-end) + decoder for token prediction.

  • Added: mel-spectrogram → Conv layers before Transformer encoder.
  • Purpose: End-to-end speech recognition / translation to text.
def whisper_forward(mel, decoder_input_ids):
    x = conv_stem(mel)  # (B, n_mels, time) -> (B, C, T')
    x = x.permute(0, 2, 1)  # (B, T', C)
    enc = encoder(x)
    dec = decoder(input_ids=decoder_input_ids, encoder_hidden_states=enc)
    return dec

6.1 BERT vs GPT-2 (bidirectional encoder vs causal decoder)

Files: modeling_bert.py, modeling_gpt2.py. Search: BertSelfAttention, GPT2Attention, causal.

# BERT: no causal mask on encoder self-attention (padding mask only)
def bert_self_attention_scores(query, key, attention_mask):
    scores = (query @ key.transpose(-1, -2)) / math.sqrt(d_head)
    if attention_mask is not None:
        scores = scores + attention_mask  # large negative for pad
    return softmax(scores, dim=-1)

# GPT-2: causal mask so position i cannot attend to j > i
def gpt2_attn_scores(query, key, causal_mask):
    scores = (query @ key.transpose(-1, -2)) / math.sqrt(d_head)
    scores = scores + causal_mask  # -inf above diagonal
    return softmax(scores, dim=-1)

6.3 LLaMA (RMSNorm + RoPE + SwiGLU)

Files: modeling_llama.py. Search: LlamaRMSNorm, apply_rotary_pos_emb, LlamaMLP.

class LlamaRMSNorm(nn.Module):
    def __init__(self, hidden_size, eps=1e-6):
        super().__init__()
        self.weight = nn.Parameter(torch.ones(hidden_size))
        self.variance_epsilon = eps

    def forward(self, hidden_states):
        input_dtype = hidden_states.dtype
        hidden_states = hidden_states.to(torch.float32)
        variance = hidden_states.pow(2).mean(-1, keepdim=True)
        hidden_states = hidden_states * torch.rsqrt(variance + self.variance_epsilon)
        return self.weight * hidden_states.to(input_dtype)

class MixtralSparseMoeBlock(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.gate = nn.Linear(config.hidden_size, config.num_local_experts, bias=False)
        self.experts = nn.ModuleList([MixtralBLockSparseTop2MLP(config) for _ in range(config.num_local_experts)])

    def forward(self, hidden_states):
        b, s, d = hidden_states.shape
        x = hidden_states.reshape(-1, d)
        router_logits = self.gate(x)
        routing_weights, expert_indices = torch.topk(
            F.softmax(router_logits, dim=-1), k=2, dim=-1
        )

        y = torch.zeros_like(x)
